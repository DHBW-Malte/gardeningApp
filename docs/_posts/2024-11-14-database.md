    USERS {
        int id PK
        string username
        string email
        string profile_picture
        datetime created_at
    }
    PLANT_CATALOG {
        int id PK
        string scientific_name
        string common_name
        string type
        string water_frequency
        string temperature_range
        string light_requirement
        string soil_type
        string fertilizer_type
    }
    PLANTS {
        int id PK
        int user_id FK
        int plant_catalog_id FK
        int garden_id FK
        string name
        datetime created_at
    }
    GARDENS {
        int id PK
        int user_id FK
        string name
        string location
        datetime created_at
    }

    USERS ||--o{ PLANTS : "owns"
    USERS ||--o{ GARDENS : "owns"
    PLANTS }o--|| PLANT_CATALOG : "refers to"
    GARDENS ||--o{ PLANTS : "contains"
